# Experi - 6: To write a program to simulate keystone correction 
# by applying geometric transformations to an image or object.

import cv2
import numpy as np
import matplotlib.pyplot as plt

# 1. Create a simple "object" (a white square on a black background)
img = np.zeros((500, 500), dtype="uint8")
cv2.rectangle(img, (100, 100), (400, 400), (255), -1)

# 2. Define coordinates for the "Distorted" (Keystone) view
# These are the 4 corners of the trapezoid we see on the wall
distorted_pts = np.float32([
    [150, 150],  # Top-left (narrower)
    [350, 150],  # Top-right (narrower)
    [50, 450],   # Bottom-left (wider)
    [450, 450]   # Bottom-right (wider)
])

# 3. Define the "Corrected" coordinates
# This is the perfect square we want to achieve
corrected_pts = np.float32([
    [100, 100], 
    [400, 100], 
    [100, 400], 
    [400, 400]
])

# 4. Calculate the Transformation Matrix and Apply it
# getPerspectiveTransform finds the math needed to map points
matrix = cv2.getPerspectiveTransform(distorted_pts, corrected_pts)
# warpPerspective applies that math to the whole image
result = cv2.warpPerspective(img, matrix, (500, 500))

# 5. Visualization - Two separate images

# Image 1: The Distorted "Keystone" Input
plt.figure(1)
plt.imshow(img, cmap='gray')
plt.title("Before: Distorted Keystone Effect")
plt.axis('off')

# Image 2: The Corrected Output
plt.figure(2)
plt.imshow(result, cmap='gray')
plt.title("After: Corrected Image")
plt.axis('off')

plt.show()
