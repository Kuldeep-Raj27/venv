# Experi - 5: To write a program to perform rotation of an S-curve 
# about a specified reference point.

import matplotlib.pyplot as plt
import numpy as np

# 1. Generate S-curve coordinates
# We use a sine wave to create the "S" shape
x = np.linspace(0, 4, 50)  # 50 points from 0 to 4
y = np.sin(x)              # Y values follow a sine curve

# 2. Rotation Parameters
xr, yr = 2, 0      # Specified Reference Point (Pivot near the center)
angle_deg = 90     # Rotation angle in degrees
theta = np.radians(angle_deg)

# 3. Perform Rotation point-by-point
x_rot = []
y_rot = []

for i in range(len(x)):
    # Step 1: Shift point so pivot is at (0,0)
    x_shift = x[i] - xr
    y_shift = y[i] - yr
    
    # Step 2: Apply rotation formula
    x_new = x_shift * np.cos(theta) - y_shift * np.sin(theta)
    y_new = x_shift * np.sin(theta) + y_shift * np.cos(theta)
    
    # Step 3: Shift back to original position
    x_rot.append(x_new + xr)
    y_rot.append(y_new + yr)

# 4. Visualization - Separate Images

# Image 1: Original S-curve
plt.figure(1)
plt.plot(x, y, 'b-', linewidth=2, label="Original S-curve")
plt.plot(xr, yr, 'ro', label="Pivot") # Mark the pivot
plt.title("Original S-Curve")
plt.grid(True)
plt.legend()
plt.axis([-5, 5, -5, 5])

# Image 2: Rotated S-curve
plt.figure(2)
plt.plot(x_rot, y_rot, 'g-', linewidth=2, label="Rotated S-curve")
plt.plot(xr, yr, 'ro', label="Pivot") # Pivot stays the same
plt.title(f"S-Curve Rotated {angle_deg} Degrees")
plt.grid(True)
plt.legend()
plt.axis([-5, 5, -5, 5])

plt.show()
