#To write a program to perform absolute rotation of a hut about a fixed
#reference point.

import matplotlib.pyplot as plt
import numpy as np

# 1. Define Hut Coordinates (X and Y separately for simplicity)
# This forms a square base and a triangle roof
x = [1, 3, 3, 1, 1, 2, 3]
y = [1, 1, 3, 3, 1, 4, 3]

# 2. Rotation Parameters
angle_deg = 45
xr, yr = 1, 1  # Fixed Reference Point (Pivot)

# Convert angle to radians
theta = np.radians(angle_deg)

# 3. Perform Rotation
x_rot = []
y_rot = []

for i in range(len(x)):
    # Step A: Shift point so pivot is at (0,0)
    x_shift = x[i] - xr
    y_shift = y[i] - yr
    
    # Step B: Apply rotation formula
    x_new = x_shift * np.cos(theta) - y_shift * np.sin(theta)
    y_new = x_shift * np.sin(theta) + y_shift * np.cos(theta)
    
    # Step C: Shift back to original reference point
    x_rot.append(x_new + xr)
    y_rot.append(y_new + yr)

# 4. Visualization (Split into 2 plots)

# Image 1: Original Hut
plt.figure(1)
plt.plot(x, y, 'b-o')
plt.plot(xr, yr, 'ro') # Mark the pivot
plt.title("Original Hut")
plt.grid(True)
plt.axis([0, 6, 0, 6])

# Image 2: Rotated Hut
plt.figure(2)
plt.plot(x_rot, y_rot, 'g-o')
plt.plot(xr, yr, 'ro') # Mark the pivot
plt.title(f"Rotated Hut ({angle_deg} degrees)")
plt.grid(True)
plt.axis([0, 6, 0, 6])

plt.show()
