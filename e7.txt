# Experi - 7: To write a program to implement Bresenham’s line generation 
# algorithm for drawing a straight line.

import matplotlib.pyplot as plt

def bresenham_line(x1, y1, x2, y2):
    # 1. Calculate constants
    dx = abs(x2 - x1)
    dy = abs(y2 - y1)
    
    # 2. Initial decision parameter
    p = 2 * dy - dx
    
    # 3. Starting points
    x, y = x1, y1
    
    x_points = []
    y_points = []
    
    # 4. Loop to calculate all points
    for i in range(dx + 1):
        x_points.append(x)
        y_points.append(y)
        
        # Increment X in every step
        x = x + 1
        
        # Check decision parameter to see if Y should increment
        if p < 0:
            p = p + 2 * dy
        else:
            y = y + 1
            p = p + 2 * dy - 2 * dx
            
    return x_points, y_points

# Coordinates for the line (x1, y1) to (x2, y2)
x_start, y_start = 2, 3
x_end, y_end = 10, 8

# Get the points from the algorithm
x_vals, y_vals = bresenham_line(x_start, y_start, x_end, y_end)

# Visualization - Split into 2 Plots

# Image 1: The calculated "Pixels" (Points)
plt.figure(1)
plt.scatter(x_vals, y_vals, color='red')
plt.title("Bresenham Algorithm: Calculated Pixel Points")
plt.xlabel("X-axis")
plt.ylabel("Y-axis")
plt.grid(True, linestyle='--')

# Image 2: The final resulting line
plt.figure(2)
plt.plot(x_vals, y_vals, 'b-o')
plt.title("Bresenham Algorithm: Resulting Straight Line")
plt.xlabel("X-axis")
plt.ylabel("Y-axis")
plt.grid(True)

plt.show()
